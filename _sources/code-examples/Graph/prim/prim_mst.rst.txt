        from DataStructures.Graph import digraph as G
        from DataStructures.Graph import prim as PRIM

        def pruebas():
            # Crear un grafo no dirigido
            my_graph = G.new_graph(0)
            my_graph = G.insert_vertex(my_graph, "Bogotá", {})
            my_graph = G.insert_vertex(my_graph, "Medellín", {})
            my_graph = G.insert_vertex(my_graph, "Cali", {})

            my_graph = G.add_edge(my_graph, "Bogotá", "Medellín", 3)
            my_graph = G.add_edge(my_graph, "Bogotá", "Cali", 1)

            # Ejecutar algoritmo de Prim desde "Bogotá"
            resultado = PRIM.prim_mst(my_graph, "Bogotá")
            print("Estructura devuelta por prim_mst:")
            print(resultado)
